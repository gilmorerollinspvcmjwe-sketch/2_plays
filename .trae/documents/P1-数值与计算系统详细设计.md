# P1 数值与计算系统详细设计 - 结合现有代码与AI能力

## 一、现有代码分析

### 1.1 已实现的计算系统

**ProjectStore 中的计算：**
- `calculateProjectHealth()`: 基于角色完成度(30%) + 剧情完成度(30%) + 团队配置(20%) + 预算(20%)
- `calculateExpectedRating()`: 基础分5 + 角色加成 + 剧情加成 + 团队加成
- `calculateExpectedRevenue()`: 市场容量 × 评分因子 × 定位因子 × 0.1
- `assessProjectRisk()`: 基于角色数、剧情数、团队人数、预算的风险评估

**OperationStore 中的计算：**
- `calculateEventImpact()`: 基于预算倍数 × 类型倍数计算预期参与人数和收入
- `simulatePoolGacha()`: 模拟抽卡，计算SSR/SR/R分布、平均保底
- `simulateOneDay()`: 随机生成收入、活跃用户、抽卡次数

**EmployeeStore 中的计算：**
- `calculateEmployeeEfficiency()`: 基础效率 - 疲劳惩罚 + 满意度加成 + 专长加成 + 性格加成
- `calculateTurnoverRisk()`: 基于疲劳度、满意度、薪资的离职风险评估

### 1.2 现有数据流

```
玩家操作 → Store更新 → 本地计算 → UI展示
                ↓
         localStorage持久化
```

**问题识别：**
1. 计算逻辑分散在各个Store中，缺乏统一的管理
2. 数值调整困难，缺乏配置化
3. 缺乏AI辅助的数据分析和建议生成
4. 联动机制尚未实现

```typescript
// src/utils/calculationEngine.ts

import { GameBalanceConfig } from '@/config/gameBalance';

export class CalculationEngine {
  private config = GameBalanceConfig;
  private cache = new Map<string, { value: any; timestamp: number }>();
  private cacheTTL = 60000; // 1分钟缓存

  // 项目评估计算
  calculateProjectHealth(project: Project): HealthResult {
    const cacheKey = `health_${project.id}`;
    const cached = this.getCache(cacheKey);
    if (cached) return cached;

    const weights = this.config.project.healthWeights;
    
    const characterCompletion = this.calculateCharacterCompletion(project.characters);
    const plotCompletion = this.calculatePlotCompletion(project.plots);
    const teamMatch = this.calculateTeamMatch(project.team);
    const budgetScore = this.calculateBudgetScore(project.budget);

    const score = 
      characterCompletion * weights.characterCompletion +
      plotCompletion * weights.plotCompletion +
      teamMatch * weights.teamMatch +
      budgetScore * weights.budget;

    const result = {
      score: Math.round(score),
      breakdown: {
        characterCompletion,
        plotCompletion,
        teamMatch,
        budgetScore
      },
      grade: this.getHealthGrade(score)
    };

    this.setCache(cacheKey, result);
    return result;
  }

  // 运营联动计算
  calculateLinkageEffect(
    sourceModule: OperationModule,
    targetModule: OperationModule,
    change: number,
    context: LinkageContext
  ): LinkageEffect {
    const rules = this.getLinkageRules(sourceModule, targetModule);
    let totalEffect = 0;

    for (const rule of rules) {
      if (this.checkCondition(rule.condition, context)) {
        const effect = this.calculateRuleEffect(rule, change, context);
        totalEffect += effect;
      }
    }

    // 应用时间衰减
    const decayedEffect = this.applyTimeDecay(
      totalEffect,
      context.delay,
      context.daysPassed
    );

    return {
      immediate: totalEffect,
      decayed: decayedEffect,
      duration: this.calculateEffectDuration(rules)
    };
  }

  // 连锁反应计算
  calculateChainReaction(
    initialEvent: OperationEvent,
    maxDepth: number = 3
): ChainReactionResult {
    const effects: ChainEffect[] = [];
    const queue: { event: OperationEvent; depth: number }[] = 
      [{ event: initialEvent, depth: 0 }];

    while (queue.length > 0) {
      const { event, depth } = queue.shift()!;
      
      if (depth >= maxDepth) continue;

      const nextEffects = this.getLinkedEffects(event);
      
      for (const effect of nextEffects) {
        effects.push({
          ...effect,
          depth,
          sourceEvent: event
        });

        // 如果效果足够大，继续传播
        if (Math.abs(effect.magnitude) > 5) {
          queue.push({
            event: this.createEffectEvent(effect),
            depth: depth + 1
          });
        }
      }
    }

    return {
      effects,
      totalImpact: this.aggregateEffects(effects)
    };
  }

  // 角色价值计算
  calculateCharacterValue(
    character: Character,
    projectMetrics: ProjectMetrics
  ): CharacterValueResult {
    const popularity = this.calculatePopularity(character, projectMetrics);
    const commercialValue = this.calculateCommercialValue(character, popularity);
    const cpValue = this.calculateCPValue(character);
    const plotCapacity = this.calculatePlotCapacity(character);

    return {
      popularity,
      commercialValue,
      cpValue,
      plotCapacity,
      overall: this.calculateOverallValue({
        popularity, commercialValue, cpValue, plotCapacity
      }),
      suggestions: this.generateValueSuggestions({
        popularity, commercialValue, cpValue, plotCapacity
      })
    };
  }

  // 剧情分析计算
  analyzePlot(plot: Plot, playerData: PlayerBehaviorData[]): PlotAnalysisResult {
    const completion = this.analyzeCompletion(plot, playerData);
    const emotionalCurve = this.analyzeEmotionalCurve(plot, playerData);
    const branchAnalysis = this.analyzeBranches(plot, playerData);
    const dropOffPoints = this.identifyDropOffPoints(completion);

    return {
      completion,
      emotionalCurve,
      branchAnalysis,
      dropOffPoints,
      quality: this.assessPlotQuality({
        completion, emotionalCurve, branchAnalysis
      }),
      suggestions: this.generatePlotSuggestions({
        completion, emotionalCurve, branchAnalysis, dropOffPoints
      })
    };
  }

  // 缓存管理
  private getCache(key: string): any | null {
    const cached = this.cache.get(key);
    if (!cached) return null;
    
    if (Date.now() - cached.timestamp > this.cacheTTL) {
      this.cache.delete(key);
      return null;
    }
    
    return cached.value;
  }

  private setCache(key: string, value: any): void {
    this.cache.set(key, { value, timestamp: Date.now() });
  }

  // 批量计算优化
  calculateBatch<T>(
    items: T[],
    calculator: (item: T) => any,
    options: { parallel?: boolean; batchSize?: number } = {}
  ): Promise<any[]> {
    const { parallel = false, batchSize = 10 } = options;

    if (parallel) {
      // 使用 Web Worker 或 requestIdleCallback 进行并行计算
      return this.calculateParallel(items, calculator, batchSize);
    }

    // 串行计算，但分批次避免阻塞UI
    return this.calculateSequential(items, calculator, batchSize);
  }

  private async calculateSequential<T>(
    items: T[],
    calculator: (item: T) => any,
    batchSize: number
  ): Promise<any[]> {
    const results: any[] = [];
    
    for (let i = 0; i < items.length; i += batchSize) {
      const batch = items.slice(i, i + batchSize);
      const batchResults = batch.map(calculator);
      results.push(...batchResults);
      
      // 让出主线程
      await new Promise(resolve => setTimeout(resolve, 0));
    }
    
    return results;
  }
}
```

```typescript
interface AIAnalysisService {

export class MockAIAnalysisService implements AIAnalysisService {
  // 基于统计方法的异常检测
  async detectAnomalies(
    metrics: TimeSeriesData[],
    sensitivity: number
  ): Promise<Anomaly[]> {
    const anomalies: Anomaly[] = [];
    
    // 计算均值和标准差
    const values = metrics.map(m => m.value);
    const mean = values.reduce((a, b) => a + b, 0) / values.length;
    const stdDev = Math.sqrt(
      values.reduce((sq, n) => sq + Math.pow(n - mean, 2), 0) / values.length
    );

    // 检测超出阈值的点
    const threshold = stdDev * (3 - sensitivity); // 灵敏度越高，阈值越低
    
    metrics.forEach((metric, index) => {
      const deviation = Math.abs(metric.value - mean);
      if (deviation > threshold) {
        anomalies.push({
          timestamp: metric.timestamp,
          value: metric.value,
          expectedValue: mean,
          deviation,
          severity: deviation > threshold * 2 ? 'high' : 'medium'
        });
      }
    });

    return anomalies;
  }

  // 简单线性回归预测
  async predictTrend(
    historicalData: TimeSeriesData[],
    forecastDays: number
  ): Promise<TrendPrediction> {
    // 计算线性趋势
    const n = historicalData.length;
    const sumX = historicalData.reduce((sum, _, i) => sum + i, 0);
    const sumY = historicalData.reduce((sum, d) => sum + d.value, 0);
    const sumXY = historicalData.reduce((sum, d, i) => sum + i * d.value, 0);
    const sumXX = historicalData.reduce((sum, _, i) => sum + i * i, 0);

    const slope = (n * sumXY - sumX * sumY) / (n * sumXX - sumX * sumX);
    const intercept = (sumY - slope * sumX) / n;

    // 生成预测
    const predictions: PredictionPoint[] = [];
    for (let i = 1; i <= forecastDays; i++) {
      const predictedValue = slope * (n + i) + intercept;
      predictions.push({
        day: i,
        predictedValue: Math.max(0, predictedValue),
        confidenceInterval: {
          lower: predictedValue * 0.9,
          upper: predictedValue * 1.1
        }
      });
    }

    return {
      trend: slope > 0 ? 'up' : slope < 0 ? 'down' : 'stable',
      slope,
      predictions,
      confidence: this.calculateConfidence(historicalData, slope)
    };
  }

  // K-means 简化版玩家分群
  async segmentPlayers(
    playerData: PlayerData[],
    segmentCount: number
  ): Promise<PlayerSegment[]> {
    // 简化实现：基于活跃度分群
    const segments: PlayerSegment[] = [];
    
    const avgSpending = playerData.reduce((sum, p) => sum + p.spending, 0) / playerData.length;
    const avgActivity = playerData.reduce((sum, p) => sum + p.activity, 0) / playerData.length;

    const segments = [

        count: playerData.filter(p => p.spending > avgSpending * 2 && p.activity > avgActivity * 1.5).length,
        characteristics: ['高付费', '高活跃', '忠诚度高']
      },
      {

        count: playerData.filter(p => p.spending < avgSpending * 0.5 && p.activity > avgActivity).length,
        characteristics: ['低付费', '高活跃', '潜在付费']
      },
      {
        name: '流失风险玩家',
        count: playerData.filter(p => p.activity < avgActivity * 0.3).length,
        characteristics: ['低活跃', '需要召回']
      }

    return segments;
  }

  // 流失风险预测
  async predictChurn(playerBehavior: PlayerBehaviorData): Promise<ChurnRisk> {
    let riskScore = 0;
    const factors: ChurnFactor[] = [];

    // 登录频率
    if (playerBehavior.daysSinceLastLogin > 7) {
      riskScore += 30;
      factors.push({ factor: '长时间未登录', weight: 30 });
    }

    // 游戏时长下降
    if (playerBehavior.playTimeDecline > 0.5) {
      riskScore += 25;
      factors.push({ factor: '游戏时长大幅下降', weight: 25 });
    }

    // 付费意愿下降
    if (playerBehavior.spendingDecline > 0.7) {
      riskScore += 20;
      factors.push({ factor: '付费意愿降低', weight: 20 });
    }

    // 社交互动减少
    if (playerBehavior.socialInteractionDecline > 0.6) {
      riskScore += 15;
      factors.push({ factor: '社交互动减少', weight: 15 });
    }

    return {
      riskScore: Math.min(100, riskScore),
      riskLevel: riskScore > 70 ? 'high' : riskScore > 40 ? 'medium' : 'low',
      factors,
      recommendedActions: this.getChurnPreventionActions(riskScore)
    };
  }

  // 智能建议生成
  async generateSmartSuggestions(
    gameState: GameState,
    goals: GameGoal[]
  ): Promise<SmartSuggestion[]> {
    const suggestions: SmartSuggestion[] = [];

    // 基于当前状态和目标生成建议
    for (const goal of goals) {
      switch (goal.type) {
        case 'increase_revenue':
          if (gameState.revenue < goal.target) {
            suggestions.push({
              priority: 'high',
              title: '优化付费点设计',
              description: '当前收入低于目标，建议推出限时礼包',
              expectedImpact: `收入提升${Math.round((goal.target - gameState.revenue) / gameState.revenue * 100)}%`,
              action: 'create_limited_pack'
            });
          }
          break;

        case 'improve_retention':
          if (gameState.retention < goal.target) {
            suggestions.push({
              priority: 'high',
              title: '优化新手体验',
              description: '次日留存率偏低，建议简化新手引导',
              expectedImpact: '留存率提升10-15%',
              action: 'optimize_tutorial'
            });
          }
          break;

        case 'boost_engagement':
          suggestions.push({
            priority: 'medium',
            title: '增加社交玩法',
            description: '玩家互动不足，建议增加公会系统',
            expectedImpact: 'DAU提升20%',
            action: 'add_guild_system'
          });
          break;
      }
    }

    // 按优先级排序
    return suggestions.sort((a, b) => {
      const priorityOrder = { high: 3, medium: 2, low: 1 };
      return priorityOrder[b.priority] - priorityOrder[a.priority];
    });
  }

  private calculateConfidence(data: TimeSeriesData[], slope: number): number {
    // 简化：数据越多，趋势越明显，置信度越高
    const dataPoints = data.length;
    const trendStrength = Math.abs(slope);
    return Math.min(0.95, (dataPoints / 100) * 0.5 + trendStrength * 0.5);
  }

  private getChurnPreventionActions(riskScore: number): string[] {
    if (riskScore > 70) {
      return ['发送召回邮件', '提供专属回归礼包', '推送个性化内容'];
    } else if (riskScore > 40) {
      return ['增加游戏内提醒', '优化核心玩法体验'];
    }
    return ['保持正常运营'];
  }
}
```

### 4.1 联动规则引擎

```typescript
// src/engine/linkageEngine.ts

export class LinkageEngine {
  private rules: Map<string, LinkageRule> = new Map();
  private effectQueue: DelayedEffect[] = [];
  private subscribers: Map<string, Set<(effect: LinkageEffect) => void>> = new Map();

  constructor() {
    this.initializeDefaultRules();
  }

  // 注册联动规则
  registerRule(rule: LinkageRule): void {
    this.rules.set(rule.id, rule);
  }

  // 触发联动
  trigger(
    sourceModule: OperationModule,
    action: string,
    value: number,
    context: LinkageContext
  ): LinkageResult {
    const triggeredRules = this.findTriggeredRules(sourceModule, action, context);
    const effects: LinkageEffect[] = [];

    for (const rule of triggeredRules) {
      const effect = this.calculateEffect(rule, value, context);
      
      if (rule.delay === 'immediate') {
        this.applyEffect(effect);
        effects.push(effect);
      } else {
        this.queueEffect(effect, rule.delay);
      }

      // 触发连锁反应
      if (effect.magnitude > rule.chainThreshold) {
        this.triggerChainReaction(effect, context);
      }
    }

    // 通知订阅者
    this.notifySubscribers(sourceModule, effects);

    return {
      triggeredRules: triggeredRules.length,
      immediateEffects: effects,
      queuedEffects: this.getQueuedEffectsCount()
    };
  }

  // 处理延迟效果队列
  processDailyEffects(): ProcessedEffect[] {
    const now = Date.now();
    const processed: ProcessedEffect[] = [];
    const remaining: DelayedEffect[] = [];

    for (const effect of this.effectQueue) {
      const daysPassed = (now - effect.queuedAt) / (1000 * 60 * 60 * 24);
      
      if (daysPassed >= effect.delayDays) {
        // 应用时间衰减
        const decayedEffect = this.applyDecay(effect, daysPassed);
        this.applyEffect(decayedEffect);
        processed.push({
          original: effect,
          decayed: decayedEffect,
          appliedAt: now
        });
      } else {
        remaining.push(effect);
      }
    }

    this.effectQueue = remaining;
    return processed;
  }

  // 订阅特定模块的联动效果
  subscribe(
    module: OperationModule,
    callback: (effect: LinkageEffect) => void
  ): () => void {
    if (!this.subscribers.has(module)) {
      this.subscribers.set(module, new Set());
    }
    this.subscribers.get(module)!.add(callback);

    // 返回取消订阅函数
    return () => {
      this.subscribers.get(module)?.delete(callback);
    };
  }

  private initializeDefaultRules(): void {
    // 卡池规则
    this.registerRule({
      id: 'gacha_rate_satisfaction',
      sourceModule: 'gacha',
      targetModule: 'playerData',
      action: 'rate_change',
      condition: (ctx) => ctx.newRate < ctx.oldRate,
      effect: (value, ctx) => ({
        targetMetric: 'satisfaction',
        change: value * GameBalanceConfig.operation.gacha.rateSatisfactionImpact,
        duration: 7
      }),
      delay: 'immediate',
      chainThreshold: 10
    });

    // 活动规则
    this.registerRule({
      id: 'event_difficulty_complaint',
      sourceModule: 'event',
      targetModule: 'incident',
      action: 'difficulty_set',
      condition: (ctx) => ctx.difficulty > 80,
      effect: (value, ctx) => ({
        targetMetric: 'incident_trigger',
        change: 1,
        incidentType: 'too_hard',
        duration: 3
      }),
      delay: 'short',
      chainThreshold: 5
    });

    // 更多规则...
  }

  private findTriggeredRules(
    sourceModule: OperationModule,
    action: string,
    context: LinkageContext
  ): LinkageRule[] {
    return Array.from(this.rules.values()).filter(rule =>
      rule.sourceModule === sourceModule &&
      rule.action === action &&
      rule.condition(context)
    );
  }

  private calculateEffect(
    rule: LinkageRule,
    value: number,
    context: LinkageContext
  ): LinkageEffect {
    const baseEffect = rule.effect(value, context);
    
    // 应用上下文修正
    const modifiedEffect = this.applyContextModifiers(baseEffect, context);
    
    return modifiedEffect;
  }

  private applyDecay(effect: DelayedEffect, daysPassed: number): LinkageEffect {
    const decayRate = 0.9; // 每天衰减10%
    const decayFactor = Math.pow(decayRate, daysPassed - effect.delayDays);
    
    return {
      ...effect,
      magnitude: effect.magnitude * decayFactor
    };
  }

  private notifySubscribers(
    module: OperationModule,
    effects: LinkageEffect[]
  ): void {
    const callbacks = this.subscribers.get(module);
    if (callbacks) {
      effects.forEach(effect => {
        callbacks.forEach(cb => cb(effect));
      });
    }
  }
}

// 单例
export const linkageEngine = new LinkageEngine();
```

```typescript
// src/composables/useLinkageTracker.ts

export function useLinkageTracker() {
  const trackedEffects = ref<TrackedEffect[]>([]);
  const unsubscribeFns: (() => void)[] = [];

  // 开始追踪特定模块的联动
  function startTracking(module: OperationModule): void {
    const unsubscribe = linkageEngine.subscribe(module, (effect) => {
      trackedEffects.value.push({
        ...effect,
        trackedAt: Date.now(),
        status: 'active'
      });

      // 限制追踪数量
      if (trackedEffects.value.length > 50) {
        trackedEffects.value = trackedEffects.value.slice(-50);
      }
    });

    unsubscribeFns.push(unsubscribe);
  }

  // 获取当前活跃的联动效果
  const activeEffects = computed(() =>
    trackedEffects.value.filter(e => e.status === 'active')
  );

  // 获取历史联动统计
  const linkageStats = computed(() => {
    const stats = {
      totalEffects: trackedEffects.value.length,
      byModule: {} as Record<OperationModule, number>,
      byDelay: {} as Record<FeedbackDelay, number>
    };

    trackedEffects.value.forEach(effect => {
      stats.byModule[effect.sourceModule] = 
        (stats.byModule[effect.sourceModule] || 0) + 1;
      stats.byDelay[effect.delay] = 
        (stats.byDelay[effect.delay] || 0) + 1;
    });

    return stats;
  });

  // 清理
  onUnmounted(() => {
    unsubscribeFns.forEach(fn => fn());
  });

  return {
    startTracking,
    activeEffects,
    linkageStats,
    trackedEffects
  };
}
```
