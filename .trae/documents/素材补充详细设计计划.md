## 1. 设计原则与规范

### 1.1 角色模板设计原则

| 原则 | 说明 | 示例 |
|------|------|------|
| **差异化** | 同类别角色要有明显区分度 | 同为总裁，冷峻vs温柔vs腹黑 |
| **标签化** | 每个角色3-5个核心标签 | ['霸道', '占有欲强', '专一'] |
| **乙女感** | 符合目标用户审美偏好 | 强调"守护"、"专一"、"反差萌" |
| **可代入** | 背景故事预留玩家位置 | "因意外与你相遇..." |
| **人气梯度** | S/A/B/C级分布合理 | S级20%、A级30%、B级30%、C级20% |

### 1.2 数据结构设计（沿用现有）

```typescript
// 角色模板结构
interface CharacterTemplate {
  id: string;              // 唯一标识: category_001
  category: CharacterCategory;
  name: string;            // 模板名称: 如"冷峻总裁"
  tags: string[];          // 标签数组
  personality: string;     // 性格描述（50-100字）
  appearance: string;      // 外貌描述（30-60字）
  clothing: string;        // 服装描述（20-40字）
  background: string;      // 背景故事（80-120字）
  voiceActor: '新人' | '资深' | '顶流';
  popularity: number;      // 1-10，影响初始人气
}
```

### 1.3 命名规范

```
文件命名: {category}.ts
ID命名: {category}_{序号:3位}
变量命名: {category}Templates
```

```typescript
  }
  
  /**
   * 随机获取指定数量的不重复角色
   */
  static getRandomCharacters(count: number, excludeIds?: string[]): CharacterTemplate[] {
    const available = excludeIds 
      ? characterTemplates.filter(t => !excludeIds.includes(t.id))
      : characterTemplates;
    
    const shuffled = [...available].sort(() => Math.random() - 0.5);
    return shuffled.slice(0, Math.min(count, shuffled.length));
  }
  
  /**
   * 获取推荐角色组合（用于新项目）
   * 策略：1个S级 + 2个A级 + 2个B级
   */
  static getRecommendedCharacterSet(): CharacterTemplate[] {
    const sRank = this.getCharactersByPopularityRange(9, 10);
    const aRank = this.getCharactersByPopularityRange(7, 8);
    const bRank = this.getCharactersByPopularityRange(5, 6);
    
    return [
      this.getRandomCharacters(1, sRank.map(t => t.id))[0],
      ...this.getRandomCharacters(2, aRank.map(t => t.id)),
      ...this.getRandomCharacters(2, bRank.map(t => t.id))
    ].filter(Boolean);
  }

  static generateComment(

    // 根据上下文选择合适的模板
    const templates = this.selectCommentTemplates(context);
    const template = templates[Math.floor(Math.random() * templates.length)];

    const content = this.fillCommentVariables(template, context);
    
    return {
```

### 6.4 模板热更新机制

```typescript
// 支持模板的热更新，无需重新打包
export class TemplateHotLoader {
  private static cache: Map<string, any> = new Map();
  
  /**
   * 从服务器加载最新模板
   */
  static async loadTemplatesFromServer(): Promise<void> {
    try {
      const response = await fetch('/api/templates/latest');
      const templates = await response.json();
      
      // 更新缓存
      this.cache.set('characters', templates.characters);
      this.cache.set('plots', templates.plots);
      this.cache.set('comments', templates.comments);
      
      // 更新本地存储
      localStorage.setItem('templates_cache', JSON.stringify(templates));
      localStorage.setItem('templates_version', templates.version);
    } catch (error) {
      console.error('模板加载失败:', error);
      // 使用缓存版本
      this.loadFromCache();
    }
  }
  
  /**
   * 检查模板更新
   */
  static async checkForUpdates(): Promise<boolean> {
    const currentVersion = localStorage.getItem('templates_version');
    const response = await fetch(`/api/templates/version?current=${currentVersion}`);
    const { hasUpdate, version } = await response.json();
    
    if (hasUpdate) {
      await this.loadTemplatesFromServer();
      return true;
    }
    
    return false;
  }
}
```
